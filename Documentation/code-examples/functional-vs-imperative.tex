List<Integer> numbers = 
	Arrays.asList( 1, 3, 4, 5, 8, 13, 15 );
List<String> eventNumbers = numbers.stream()
        .filter(n -> n % 2 == 0)
        .map(n -> String.valueOf(n))
        .collect(Collectors.toList());
doSomething(eventNumbers);
\end{javacode}
