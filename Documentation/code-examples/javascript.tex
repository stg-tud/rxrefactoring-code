function validateNumber() {
    var x, text;
    x = document.getElementById("numb").value;
    if ( x == ""){
            text = "";
    } else if (isNaN(x)) {
        text = "Not a number";
    } else if (x < 1 || x > 10) {
        text = "Number out of Range";
    } else {
        text = "Valid number";
    }
    document.getElementById("validationMessage")
    	.innerHTML = text;
}
